326  std::string content = strtrim(data);
327 
328  if (typ==RSJ_OBJECT || typ==RSJ_UNKNOWN) {
329  // parse as object:

332  std::vector<std::string> nvPairs = split_RSJ_array (content);
333  for (unsigned int a=0; a<nvPairs.size(); ++a) {
334  std::size_t assignmentPos = nvPairs[a].find (RSJobjectassignment);
335  object.insert (make_pair(
336  strip_outer_quotes (nvPairs[a].substr (0,assignmentPos) ) ,
337  RSJresource (strtrim (nvPairs[a].substr (assignmentPos+1) ) )
338  ) );
339  }
340  if (object.size() > 0) {
341  type = RSJ_OBJECT;
342  return;
343  }
344  }
345  }
346 
347  if (typ==RSJ_ARRAY || typ==RSJ_UNKNOWN) {
348  // parse as array
349  content = strtrim (strtrim (content, "[", 1, STRTRIM_L ), "]", 1, STRTRIM_R );
350  if (content.length() != data.length()) { // a valid array
351  std::vector<std::string> nvPairs = split_RSJ_array (content);
352  for (unsigned int a=0; a<nvPairs.size(); ++a)
353  array.push_back (RSJresource (strtrim (nvPairs[a]) ) );
354  if (array.size() > 0) {
355  type = RSJ_ARRAY;
