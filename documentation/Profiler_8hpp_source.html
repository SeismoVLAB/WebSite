102  void WriteProfile(const ProfileResult& result){
103  if (m_ProfileCount++ > 0)
104  m_OutputStream << ",";
105 
106  std::string name = result.Name;
107  std::replace(name.begin(), name.end(), '"', '\'');
108 
109  //The JSON file format to be opened with Google-Chrome
110  m_OutputStream << "{";
111  m_OutputStream << "\"cat\":\"function\",";
112  m_OutputStream << "\"dur\":" << (result.End - result.Start) << ',';
113  m_OutputStream << "\"name\":\"" << name << "\",";
114  m_OutputStream << "\"ph\":\"X\",";
115  m_OutputStream << "\"pid\":\"" << result.ThreadID << "\",";
116  m_OutputStream << "\"tid\":" << rank << ",";
117  m_OutputStream << "\"ts\":" << result.Start;
118  m_OutputStream << "}";
119 
120  //Flush in case the program abruptly stops.

122  }
